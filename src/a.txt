INT f2(INT x, INT y ) 
BEGIN 
	INT z := 0;
	WHILE ((z <= x*x - y*y))
	BEGIN
		z := z + 1;
	END
	RETURN z; 
END 
INT MAIN() 
BEGIN
	INT x;
	READ(x, "A41.input");
	INT y;
	READ(y, "A42.input");
	INT z;
	IF (x<=2 AND y>= 4)
	BEGIN
		z := f2(x+2,y-4);
	END
	ELSE
		z := f2(x,y) + f2(y,x);
	WRITE (z, "A4.output"); 
END
